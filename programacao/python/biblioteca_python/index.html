    Número aleatórios¶

    Desde a versão 3.6, o Pyhton passou a incorporar o módulo secrets. Esse módulo pode ser usado, em substituição do módulo random, para a geração de números aleatórios criptograficamente seguros. A principal diferença é que esse novo módulo utiliza, sempre, a fonte mais segura de aleatorização disponível no sistema.

    import secrets
import string

#          A-Z, a-z             + 0-9
alphabet = string.ascii_letters + string.digits

# Seleciona um elmento da lista alphabet
password = ''.join(secrets.choice(alphabet) for i in range(10))

# Retorna um número inteiro, positivo, abaixo de 20
password = secrets.randbelow(20)

# Retorna um inteiro, positivo, com 7 bits
password = secrets.randbits(7)

# Retorna uma cadeia de bytes contendo 10 bytes
token = secrets.token_bytes(10)

# Retorna uma cadeia de caracteres contendo 10 bytes
# representados em hexadecimal
token2 = secrets.token_hex(10)

    O módulo random ainda está disponível, e pode ser usado em situações nas quais a robustez oferecida pelo módulo secrets não for necessária.

    import random
import string

#          A-Z, a-z             + 0-9
alphabet = string.ascii_letters + string.digits

# Inicializa o gerador de números pseudo-aleatórios com uma semente
# igual ao horário atual
random.seed()

# Retorna um número de ponto flutuante, [0.00 e 1.00[
numero = random.random()
# Seleciona um elmento da lista alphabet
password = ''.join(random.choice(alphabet) for i in range(10))

# Retorna um número inteiro, positivo, abaixo de 20
password = random.randint(0, 20)

# Retorna um inteiro, positivo, com 7 bits
password = random.getrandbits(7)

# Embaralha uma lista
mensagem = list('Mensagem')
print(mensagem)  # ['M', 'e', 'n', 's', 'a', 'g', 'e', 'm']
random.shuffle(mensagem)
print(mensagem)  # ['m', 'n', 'e', 'e', 'M', 'a', 's', 'g']

    from datetime import datetime

now = datetime.now()

# dd/mm/YY
d1 = now.strftime("%d/%m/%Y")
print("d1 =", d1)       # d1 = 20/03/2022

# Mês textual, dia e ano
d2 = now.strftime("%B %d, %Y")
print("d2 =", d2)       # d2 = March 20, 2022

# mm/dd/y
d3 = now.strftime("%m/%d/%y")
print("d3 =", d3)       # d3 = 03/20/22

# Abreviação do mês, dia e ano
d4 = now.strftime("%b-%d-%Y")
print("d4 =", d4)       # d4 = Mar-20-2022

# H:M:S
d5 = now.strftime("%H:%M:%S")
print("d5 =", d5)       # d5 = 22:35:39

# Estamos no ano de 2022, mês de 3
print(f"Estamos no ano de {now.year}, mês {now.month}")
