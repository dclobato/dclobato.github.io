/**
 * @brief Lê um número inteiro da entrada padrão.
 *
 * Esta função solicita ao usuário a entrada de um valor inteiro através
 * do prompt fornecido. A função continuará solicitando a entrada até que
 * um valor inteiro válido seja fornecido, dentro dos limites do tipo int.
 *
 * @param prompt Mensagem exibida ao usuário solicitando a entrada (pode
 *        ser NULL)
 * @return O valor inteiro lido da entrada padrão
 */
int lerInt(const char *prompt)
{
  char *linha = NULL;
  char *p     = NULL;
  long  valor = 0L;

  while (1)
  {
    if (prompt != NULL)
    {
      printf("%s", prompt);
    }
    linha = lerLinha(NULL);
    valor = strtol(linha, &p, 10);
    if ((p == linha) || (errno == ERANGE) || (*p && *p != '\n'))
    {
      free(linha);
      continue;
    }
    if (valor < INT_MIN || valor > INT_MAX)
    {
      free(linha);
      continue;
    }
    break;
  }
  free(linha);
  return (int) valor;
}

/**
 * @brief Lê uma linha de texto da entrada padrão.
 *
 * Esta função lê uma linha completa de texto da entrada padrão
 * (stdin), até encontrar um caractere de nova linha ou EOF. A memória
 * para armazenar a linha é alocada dinamicamente.
 *
 * @param prompt Mensagem exibida ao usuário solicitando a entrada (pode
 *        ser NULL)
 * @return Ponteiro para a string lida (deve ser liberada pelo chamador
 *         usando free())
 */
char *lerLinha(const char *prompt)
{
  char *line = NULL;
  if (prompt != NULL)
  {
    printf("%s", prompt);
  }
  line = lerLinhaArquivo(stdin);
  return line;
}

/**
 * @brief Lê uma linha de texto de um arquivo especificado.
 *
 * Esta função lê uma linha completa de texto do arquivo especificado,
 * até encontrar um caractere de nova linha ou EOF. A memória para
 * armazenar a linha é alocada dinamicamente. A função gerencia
 * automaticamente o redimensionamento do buffer se necessário.
 *
 * @param arquivo Ponteiro para o arquivo de onde a linha será lida
 * @return Ponteiro para a string lida (deve ser liberada pelo chamador
 *         usando free()), ou NULL se EOF for encontrado sem ler nenhum
 *         caractere
 */
char *lerLinhaArquivo(FILE *arquivo)
{
  char *       line  = NULL;
  char *       nline = NULL;
  unsigned int n     = 0;
  unsigned int size  = 4096;
  int          ch    = 0;

  line = (char *) malloc(size + 1);
  if (line == NULL)
  {
    printf("Sem memoria\n");
    exit(1);
  }
  while ((ch = getc(arquivo)) != '\n' && ch != EOF)
  {
    if (n == size)
    {
      size += size;
      nline = (char *) malloc(size + 1);
      if (nline == NULL)
      {
        free(line);
        printf("Sem memoria\n");
        exit(1);
      }
      strncpy(nline, line, n);
      free(line);
      line = nline;
    }
    line[n++] = (char) ch;
  }
  if (n == 0 && ch == EOF)
  {
    free(line);
    return (NULL);
  }
  line[n] = '\0';
  nline   = (char *) malloc(n + 1);
  if (nline == NULL)
  {
    free(line);
    printf("Sem memoria");
    exit(1);
  }
  strcpy(nline, line);
  free(line);
  return nline;
}

    #include <ctype.h>
#include <string.h>

void trimString(char *str) {
    int start = 0, end = strlen(str) - 1;

    // Remove espaços no início
    while (isspace((unsigned char)str[start]))
        start++;

    // Remove espaços no final
    while ((end >= start) && isspace((unsigned char)str[end]))
        end--;

    // Move os caracteres para o início
    memmove(str, str + start, end - start + 1);
    str[end - start + 1] = '\0';
}

/**
 * @brief Inicializa o gerador de números aleatórios.
 *
 * Esta função inicializa o gerador de números pseudoaleatórios
 * utilizando o tempo atual como semente. Deve ser chamada uma única
 * vez no início do programa antes de qualquer chamada às funções
 * RandomInteger ou RandomReal.
 */
void Randomize(void)
{
  srand((int) time(NULL));
}

/**
 * @brief Gera um número inteiro aleatório dentro de um intervalo especificado.
 *
 * Esta função retorna um número inteiro pseudoaleatório uniformemente
 * distribuído no intervalo [low, high], inclusive. Requer que a função
 * Randomize() tenha sido chamada previamente para inicializar o gerador
 * de números aleatórios.
 *
 * @param low Limite inferior do intervalo
 * @param high Limite superior do intervalo
 * @return Um número inteiro aleatório entre low e high, inclusive
 */
int RandomInteger(int low, int high)
{
  int k;
  double d;

  d = (double) rand() / ((double) RAND_MAX + 1);
  k = (int) (d * (high - low + 1));
  return (low + k);
}

/**
 * @brief Gera um número real aleatório dentro de um intervalo especificado.
 *
 * Esta função retorna um número de ponto flutuante pseudoaleatório
 * uniformemente distribuído no intervalo [low, high). Note que o limite
 * superior (high) não está incluído no intervalo de valores possíveis.
 * Requer que a função Randomize() tenha sido chamada previamente para
 * inicializar o gerador de números aleatórios.
 *
 * @param low Limite inferior do intervalo (incluído)
 * @param high Limite superior do intervalo (não incluído)
 * @return Um número real aleatório entre low (inclusive) e high (exclusive)
 */
double RandomReal(double low, double high)
{
  double d;

  d = (double) rand() / ((double) RAND_MAX + 1);
  return (low + d * (high - low));
}
